library(rpart)


accuracy_tune <- function(fit) {
  predict_att <- predict(fit, testdata, type = 'class')
  table_mat <- table(testdata$Attrition, predict_att)
  accuracy_Test <- sum(diag(table_mat)) / sum(table_mat)
  accuracy_Test
}

control <- rpart.control(minsplit = 4,minbucket = round(5 / 3),maxdepth = 3,cp = 0)
tune_fit <- rpart(Attrition~., data = traindata, method = 'class', control = control)
accuracy_tune(tune_fit)
